class LongestSubstringLength {
    public int lengthOfLongestSubstring(String s) {
        HashMap<Character,Integer> map =new HashMap<>();
        int l=0;
        int maxl=0;
        for(int r=0;r<s.length();r++){
            char ch=s.charAt(r);
            if(map.containsKey(ch) && map.get(ch)>=l){
                l=map.get(ch)+1;
            }
            map.put(ch,r);
            maxl=Math.max(r-l+1,maxl);
        }
        return maxl;
        
    }
}
