class getDescentPeriods {
    public long getDescentPeriods(int[] prices) {
        long c=1;
        long len=1;
        for(int i =0;i<prices.length-1;i++){
            if(prices[i]-prices[i+1]==1){
                len++;
            }
            else{
                len=1;
            }
            c+=len;
        }
        return c;
    }
}
