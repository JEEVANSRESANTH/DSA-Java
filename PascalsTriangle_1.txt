class PascalsTriangle_1 {
    public List<List<Integer>> generate(int numRows) {
        List<List<Integer>> arl=new ArrayList<>();
        for(int i=0;i<numRows;i++){
            int v=1;
            List<Integer> res =new ArrayList<>();
            for(int j=0;j<=i;j++){
                res.add(v);
                v=v*(i-j)/(j+1);
            }
            arl.add(res);
        }
        return arl;
    }
}
